import numpy as np
tokenizer = tensorflow.keras.preprocessing.text.Tokenizer()
tokenizer.fit_on_texts((x.content for x in TextData().generate()))

kl = list(tokenizer.word_index.keys())
kl_tfidf = tokenizer.texts_to_matrix(kl, mode='tfidf')
